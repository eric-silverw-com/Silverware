    VAR
      WhseInternalPutAwayHeader@1002 : Record 7331;
      WhseInternalPutAwayLine@1000 : Record 7332;
      ReleaseWhseInternalPutAway@1001 : Codeunit 7316;
    BEGIN
      WhseInternalPutAwayLine.COPY(Rec);

    END;
