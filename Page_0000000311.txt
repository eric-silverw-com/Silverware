    LOCAL PROCEDURE ErrorsExistOnCurrentBatch@2(ShowError@1002 : Boolean) : Boolean;

      ErrorMessage.SetContext(IntrastatJnlBatch);
      EXIT(ErrorMessage.HasErrors(ShowError));
    END;

    END;
