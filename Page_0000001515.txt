    VAR
      WorkflowStep@1001 : Record 1502;
      WorkflowEvent@1000 : Record 1520;
    BEGIN
      WorkflowStep.GET(GETRANGEMAX("Workflow Code"),GETRANGEMAX("Parent Event Step ID"));
