      AllObjWithCaption@1000 : Record 2000000058;

    LOCAL PROCEDURE RunReportOrPageLink@2();
    VAR
      AllObjWithCaption@1000 : Record 2000000058;
    BEGIN
      IF ("Object Type" = 0) OR ("Object ID" = 0) THEN
        EXIT;
      IF "Object Type" = AllObjWithCaption."Object Type"::Page THEN
        PAGE.RUN("Object ID")
      ELSE
        REPORT.RUN("Object ID");
    END;
