    LOCAL PROCEDURE CallSetup@1();
    VAR
      RecordRefVariant@1001 : Variant;
      RecordRef@1000 : RecordRef;
    BEGIN
      IF NOT SetupActive THEN

      RecordRef.GET("Record ID");

      DELETE;
