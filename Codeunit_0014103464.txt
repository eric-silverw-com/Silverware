    [EventSubscriber(Page,256,OnBeforeActionEvent,VoidPayments,Skip,Skip)]
    LOCAL PROCEDURE BeforePaymentJournalVoid@1000000007(VAR Rec@1000000000 : Record 81);
    VAR
      GenJnlLine@1000000001 : Record 81;
      EasyPDFPrimaryKey@1000000003 : Text;
      EasyPDFQueue@1000000002 : Record 14103463;
    BEGIN
      // Check for lines where a remittance has been delivered - if any exist confirm void
      GenJnlLine.RESET;
      GenJnlLine := Rec;
      GenJnlLine.SETRANGE("Journal Template Name",Rec."Journal Template Name");
      GenJnlLine.SETRANGE("Journal Batch Name",Rec."Journal Batch Name");
      GenJnlLine.SETRANGE("Index Entry",TRUE);  // Easy PDF uses this field to indicate remittance
      IF GenJnlLine.FIND('-') THEN
        IF NOT CONFIRM(BeforePaymentJournalVoidText) THEN
          ERROR(BeforePaymentJournalVoidError);

      // Remove undelivered entries from the Easy PDF remittance batch
      GenJnlLine.RESET;
      GenJnlLine := Rec;
      GenJnlLine.SETRANGE("Journal Template Name",Rec."Journal Template Name");
      GenJnlLine.SETRANGE("Journal Batch Name",Rec."Journal Batch Name");
      IF GenJnlLine.FIND('-') THEN
        REPEAT
          EasyPDFPrimaryKey := GenJnlLine."Journal Template Name" + ',' + GenJnlLine."Journal Batch Name" + ',' + FORMAT(GenJnlLine."Line No.");
          EasyPDFQueue.RESET;
          EasyPDFQueue.SETRANGE(PrimaryKey, EasyPDFPrimaryKey);
          EasyPDFQueue.SETRANGE(Processed, FALSE);
          EasyPDFQueue.DELETEALL(TRUE);
        UNTIL GenJnlLine.NEXT = 0;

      COMMIT;
    END;

      GenJnlLine@1000000004 : Record 81;
      PrimaryKey@1000000003 : Text;
      EasyPDFQueue@1000000002 : Record 14103463;
      Confirmed@1000000001 : Boolean;
    BEGIN
      // If Easy PDF batch queue entries exist confirm batch delete
      Confirmed := FALSE;
      GenJnlLine.RESET;
      GenJnlLine := Rec;
      GenJnlLine.SETRANGE("Journal Template Name",Rec."Journal Template Name");
      GenJnlLine.SETRANGE("Journal Batch Name",Rec."Journal Batch Name");

          EasyPDFQueue.SETRANGE(PrimaryKey, PrimaryKey);

                ERROR(BeforePaymentJournalPostError);

            EasyPDFQueue.DELETE(TRUE);
          END;
        UNTIL GenJnlLine.NEXT = 0;

      COMMIT;
    END;

    LOCAL PROCEDURE AfterExportPayment@1000000011(VAR Rec@1000000000 : Record 81);
    VAR
      GenJournalLineRec@1000000002 : Record 81;
      ReportParameters@1000000003 : Text;
      EasyPDF@1000000004 : Codeunit 37062300;
      EasyPDFCustomization@1000000006 : Codeunit 14103460;
      EasyPDFDocument@1000000007 : Record 14103461;
      PrimaryKey@1000000001 : Text;
      BankAccountNo@1000000005 : Code[20];
    BEGIN
      EasyPDF.GetDocumentSetup(EasyPDFDocument, 'REMITTANCE ADVICE');
      IF NOT EasyPDFDocument.BatchOnPost THEN
        EXIT;

      PrimaryKey :=
        Rec."Journal Template Name" + ',' +
        Rec."Journal Batch Name" + ',' +
        FORMAT(Rec."Line No.");

      IF Rec."Account Type" = Rec."Account Type"::"Bank Account" THEN
        BankAccountNo := Rec."Account No.";
      IF Rec."Bal. Account Type" = Rec."Bal. Account Type"::"Bank Account" THEN
        BankAccountNo := Rec."Bal. Account No.";

      GenJournalLineRec.COPY(Rec);
      GenJournalLineRec.SETRANGE("Journal Template Name", GenJournalLineRec."Journal Template Name");
      GenJournalLineRec.SETRANGE("Journal Batch Name", GenJournalLineRec."Journal Batch Name");
      GenJournalLineRec.SETRANGE("Line No.", GenJournalLineRec."Line No.");
      ReportParameters := EasyPDF.NewReportParameters(REPORT::"Print Electronic Payments");
      ReportParameters := EasyPDF.AddReportParameter(ReportParameters, 0, 'EasyPDFInvoked', 'true');
      ReportParameters := EasyPDF.AddReportParameter(ReportParameters, 0, 'EasyPDFBankAccountNo', BankAccountNo);
      ReportParameters := EasyPDF.AddReportParameter(ReportParameters, 1, 'Gen. Journal Line', GenJournalLineRec.GETVIEW);

      EasyPDF.QueueBatchEx(
        'REMITTANCE ADVICE',
        FORMAT(Rec."Line No."),
        PrimaryKey,
        EasyPDFCustomization.GenerateBatchId('REMITTANCE ADVICE'),
        ReportParameters);
    END;

    [EventSubscriber(Page,22,OnAfterActionEvent,Action14103465,Skip,Skip)]
    LOCAL PROCEDURE CustomerListBatchCustomerStatement@1000000060(VAR Rec@1000000000 : Record 18);
    VAR
      EasyPDF@1000000006 : Codeunit 37062300;
      EasyPDFCustomization@1000000005 : Codeunit 14103460;
      EasyPDFQueue@1000000004 : Record 14103463;
      EasyPDFBatchReportId@1000000003 : Integer;
      EasyPDFDocument@1000000002 : Record 14103461;
      EasyPDFObject@1000000001 : Record 2000000001;
      EasyPDFParameters@1000000007 : Text;
      EASYPDFERR01@1000000009 : TextConst 'ENU=Report not found (%1)';
      EASYPDFERR02@1000000008 : TextConst 'ENU=Easy PDF Setup for Customer Statement batch generation does not match this report (%1)';
    BEGIN
      // Check Setup to make sure this is the correct report
      EasyPDF.GetDocumentSetup(EasyPDFDocument, 'CUSTOMER STATEMENT');

      // To ensure that there is not a mismatch between the batch generation report and the printed report
      // we require that the name of the batch generation report contains the ID of the printed report.

      EasyPDFBatchReportId := EasyPDFDocument.BatchGenerationReportId;
      IF NOT EasyPDFObject.GET(EasyPDFObject.Type::Report, '', EasyPDFBatchReportId) THEN
        ERROR(EASYPDFERR01, EasyPDFBatchReportId);
      IF STRPOS(EasyPDFObject.Name, FORMAT(EasyPDFDocument."Report ID")) = 0 THEN
        ERROR(EASYPDFERR02, EasyPDFDocument."Report ID");

      // Run the request page for the report and collect parameters
      EasyPDFParameters := REPORT.RUNREQUESTPAGE(EasyPDFBatchReportId);
      IF EasyPDFParameters = '' THEN
        EXIT;

      // Execute the report - with invocation by EasyPDF for batch generation
      EasyPDFParameters := EasyPDF.SetReportParameter(EasyPDFParameters, 'EasyPDFBatchGeneration', 'true');
      REPORT.EXECUTE(EasyPDFBatchReportId, EasyPDFParameters);

      // Show the batch queue if batch entries exist
      EasyPDFQueue.RESET;
      EasyPDFQueue.SETRANGE(Processed, FALSE);
      EasyPDFQueue.SETRANGE(BatchId, EasyPDFCustomization.GenerateBatchId('CUSTOMER STATEMENT'));
      IF EasyPDFQueue.COUNT > 0 THEN
        PAGE.RUN(PAGE::"EasyPDF Batch Queues", EasyPDFQueue);
    END;

    VAR
      EasyPDFQueue@1000000001 : Record 14103463;
      EasyPDFCustomization@1000000005 : Codeunit 14103460;
      EasyPDF@1000000006 : Codeunit 37062300;
      LedgerEntry@1000000004 : Record 25;
      BatchId@1000000003 : Text[80];
      Count@1000000002 : Integer;
    BEGIN
      // TODO: It is not currently possible to identify selected entries in an event subscriber
      // The code below is not invoked but is included here to use when the functionality is added to NAV
      EXIT;

      Count := 0;
      //CurrPage.SETSELECTIONFILTER(LedgerEntry);
      IF LedgerEntry.FIND('-') THEN
      BEGIN

        BatchId := EasyPDFCustomization.GenerateBatchId('VENDOR LEDGER ENTRIES');
        REPEAT
          BEGIN
            CASE LedgerEntry."Document Type" OF
              LedgerEntry."Document Type"::Invoice:
                BEGIN
                  EasyPDF.QueueBatch('POSTED PURCHASE INVOICE', LedgerEntry."Document No.", BatchId);
                  Count += 1;
                END;
              LedgerEntry."Document Type"::Payment:
                BEGIN
                  EasyPDF.QueueBatchEx('VENDOR PAYMENT RECEIPT', LedgerEntry."Document No.", FORMAT(LedgerEntry."Entry No."), BatchId, '');
                  Count += 1;
                END;

      IssuedReminder@1000000003 : Record 297;
      PostedServiceInvoice@1000000002 : Record 5992;
      PostedServiceCreditMemo@1000000001 : Record 5994;
    BEGIN
      CASE LedgerEntry."Document Type" OF

        LedgerEntry."Document Type"::Invoice:
          BEGIN
            IF PostedSalesInvoice.GET(LedgerEntry."Document No.") THEN
              EXIT('POSTED SALES INVOICE');
            IF PostedServiceInvoice.GET(LedgerEntry."Document No.") THEN
              EXIT('POSTED SERVICE INVOICE');

        LedgerEntry."Document Type"::"Credit Memo":

              EXIT('POSTED SALES CREDIT MEMO');

              EXIT('POSTED SERVICE CREDIT MEMO');

              EXIT('ISSUED FINANCE CHARGE MEMO');

        LedgerEntry."Document Type"::Reminder:

      EXIT('');
