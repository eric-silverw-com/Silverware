                 END;

                 IF ApprovalEntry.FIND('-') THEN BEGIN
                   REPEAT
                     AddApprovalEntry(ApprovalEntry,WorkflowsCounter);
                   UNTIL ApprovalEntry.NEXT = 0;
                 END;
               END;
