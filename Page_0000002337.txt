    LOCAL PROCEDURE GetWindowsLanguageNameFromID@12(ID@1000 : Integer) : Text;

      WindowsLanguage@1001 : Record 2000000045;

      IF WindowsLanguage.GET(ID) THEN
        EXIT(WindowsLanguage.Name);
    END;
