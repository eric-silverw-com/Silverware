    LOCAL PROCEDURE SetCalculatedFields@6();
    VAR
      AccountingPeriod@1001 : Record 50;
      GeneralLedgerSetup@1002 : Record 98;
      GraphIntegrationRecord@1003 : Record 5451;
      GraphMgtCompanyInfo@1000 : Codeunit 5473;
    BEGIN

    END;
