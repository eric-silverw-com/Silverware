    PROCEDURE GetEDMXML@1() : Text;
    VAR
      EDMDefinitionInStream@1000 : InStream;
      EDMText@1001 : Text;
    BEGIN
      CALCFIELDS("Edm Xml");
      IF NOT "Edm Xml".HASVALUE THEN

      "Edm Xml".CREATEINSTREAM(EDMDefinitionInStream,TEXTENCODING::UTF8);
      EDMDefinitionInStream.READ(EDMText);
