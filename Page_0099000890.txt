                       IF MachineCenter."Capacity (Effective)" <> 0 THEN
                         CapacityEfficiency := ROUND(MachineCenter."Prod. Order Need (Qty.)" / MachineCenter."Capacity (Effective)" * 100,0.1)
                       ELSE
                         CapacityEfficiency := 0;
