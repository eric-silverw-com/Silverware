    VAR
      OriginalFilterGroup@1000 : Integer;
    BEGIN
      OriginalFilterGroup := FILTERGROUP;

      FILTERGROUP(OriginalFilterGroup);
    END;

    LOCAL PROCEDURE RunReportOrPageLink@2();
    VAR
      AllObjWithCaption@1000 : Record 2000000058;
    BEGIN

        EXIT;

        PAGE.RUN("Object ID")
      ELSE
        REPORT.RUN("Object ID");
    END;
